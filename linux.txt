	---Работа с каталогами---
ls - выводит список содержимого каталога [list - список];
	"ls -l путь" - вывод списка содержимого с доп. информацией;
pwd - вывести рабочий каталог [print working directory - вывести рабочий каталог];
mkdir - создать каталог [make directory - создать каталог] {"mkdir путь/имя"};
cd - изменить текущий каталог [change directory - изменить каталог] {"cd путь"};
	---Работа с файлами и каталогами---
rm - удалить (ссылки на) файлы или каталоги [remove - удалять] {"rm путь"};
	"rm путь -r" - удалить папку;
cp - копировать файл или каталог [copy - копировать] {cp путь_источника путь_копии/имя};
	"cp cp путь_источника путь_копии/имя -r" - копировать папку со всем содержимым;
mv - переместить (и/или переименовать) файл или каталог [move - перемещать] {mv путь_источника путь_назначения};
locate - поиск в БД имён файлов и вывод всех имён, соответствующих искомой строке {locate критерий};
find - поиск файлов согласно заданным критериям в указанном каталоге {find критерий};
	---Работа с файлами---
touch - изменяет времена, ассоциированные с файлом {"touch путь"};
	"touch test.txt" - создать пустой файл (если такого файла не существует);
cat - создавать и объединять содержимое файлов, а также считать данные в указанном файле и отображать их на экране [concatenate - конкатенация, сцеплять] {"cat путь"};
	"cat путь_к_файлу1 путь_к_файлу2 > путь_к_файлу3" - обьединить содержимое двух файлов в один;
	"cat > путь_к_файлу" - записать все введённые значения с клавиатуры в файл;
	"cat /etc/passwd" - вывод списка всех учётных записей пользователей (a2:x:1001:1001:,,,:/home/a2:/bin/bash - имя:пароль:UID(ID пользователя):GID (ID основной группы):GECOS(доп.инфо.):Домашний каталог:Абсолютный путь к оболочке);
	"cat /etc/group" - вывод списка всех групп (a2:x:1001: - имя:пароль:ID группы:Список пользователей);
	---Работа с экраном---
clear - очистить экран {clear};
	---Работа с процессами---
ps - вывод списка процессов [process status - статус процесса] {ps};
	"ps -all" - вывод всех процессы на экран;
kill - отправить сигнал процессу {kill параметр PID};-
	"kill -9 PID" - немедленно завершить процесс под номером PID;
	---Работа с учётными записями---
su - запуск командной оболочки от имени другого пользователя [substitute user - заменить пользователя] {su имя_уч_записи};
	"su" - запуск оболочки от имени root;
sudo -  выполнить команду от имени другого пользователя [substitute user and do - заменить пользователя и сделать] {sudo имя_уч_записи команда};
	"sudo команда" - выполнить команду от имени root;
adduser - создать новую учётную запись в системе {adduser имя_уч_записи};
deluser - удалить учётную запись из системы. Домашняя папка не удаляеться {deluser имя_уч_записи};
	"deluser --remove-home имя" - удалить учётную запись с домашней папкой;
addgroup - создать группу в системе {addgroup имя_группы};
	"addgroup имя_уч_записи имя_группы" - добавить учётную запись в группу;
groupadd - создать группу в системе. Утилита {groupadd имя_группы};
delgroup - удалить группу из системы [delete group - удалить группу] {delgroup имя_группы};
	"delgroup имя_уч_записи имя_группы" - удалить учётную запись из группы;
groupdel - удалить группу из системы. Утилита [group delete - группу удалить] {groupdel имя_группы};
passwd - изменить пароль учётной записи [password - пароль] {passwd имя_уч_записи};
	"passwd -d имя_уч_записи" - удалить пароль для учётной записи;
exit - выйти из учётной записи {exit};
	---Работа с правами доступа---
chmod - изменить права доступа к файлу или каталогу [change mode - изменить режим] {chmod права_доступа путь_к_файлу};
	права_доступа: r - читать [read - читать], w - изменять [write - записывать], x - выполнять [execute - выполнить];
	права_доступа: 0 (---), 1 (--x), 2 (-w-), 3 (-wx), 4 (r--), 5 (r-x), 6 (rw-), 7 (rwx);
	"chmod 750 путь_к_файлу" - владелец может чтитать, записывать(изменять) и выполнять; группа владельца может читать и выполнять; остальным ничего не разрешено;
chown - смена владельца и группы файла или каталога [change owner - сменить владельца] {chown имя_уч_записи1:имя_уч_записи2 путь};
	---Работа с ОС---
uname - отобразить информацию об операционной системе [UNIX name - UNIX имя] {uname -a};
poweroff - завершение работы системы;
reboot - перезагрузка системы;
	---Работа со временем---
date - вывод текущей даты {date};
	"date +%a" - вывод только название текущего дня недели;
	---Операторы---
> - перенаправить стандартный вывод {команда1 > команда2};
	"ls -l > путь_к_файлу" - перенаправление вывода команды ls в файл (файл существует - перезапись, файл не существует - создание с содержимым);
	"> путь_к_файлу" - создать пустой файл (если такого файла не существует);
>> - перенаправить стандартный вывод (добавить вывод команды к содержимому файла) {команда1 >> команда2}; 
	"ls -l >> путь_к_файлу" - добавить к содержимому файла вывод команды ls;
	">> путь_к_файлу" - создать пустой файл;
2> - перенаправить вывод ошибок {команда1 2> команда2};
	"команда_с_ошибкой 2> путь_к_файлу" - запись в файл ошибку выполнения команды;
	"команда_с_возможной_ошибкой 2> /div/null" - вывод всех ошибок в битоприёмник (безошибочное выполнение команды выведиться на экран);
?> - перенаправить стандартный вывод и вывод ошибок {команда1 &> команда2};
	"команда &> путь_к_файлу" - запись в файл ошибку выполнения команды и стандартное выполнение команды;
	---Конвееры---
| - связать стандартный вывод одной команды со стандартным вводом другой {команда1 | команда2};
	"ls -l путь | less" - построчное отображение вывода;
	---Фильтры---
grep - поиск и вывод строк, соответствующих шаблону {grep шаблон путь_к_файлу};
	"ls -l / | grep шаблон" - вывод строк из команды ls, которые соответствуют шаблону;
	---Справочная информация---
man - просмотреть справочное руководство для команды [manual - инструкция] {man команда};
	---Обозначения---
~ - путь домашнего каталога;
/ - путь корневого каталога;
